时间范围选择器

定义 & 应用场景

该控件适用于在一周范围内选择非连续的多段时间

展示如下：

timerange 依赖 drop 插件的支持，请使用前加载好相关依赖

		
<div class="ndp-timerange-wrapper" name="plain-tab"></div>
		
	
		
var the = $(".ndp-timerange-wrapper").timerange({
	lan:"en",// 中英文 "en", "cn"
	timezone:0  //下拉菜单里面的时区，序号 默认选第一个
});


//调用这个函数 获取选择的数据，当面第一个按钮，点击会调用这个函数
the.getVal();// 获得选取的数据   数据格式如下
//[{},{}]  {timeStart:"开始时间"，timeEnd:结束时间,day:星期几,zone:{}} 		
	

//上面第二个按钮点击会调用这个函数，设置数据
//arr  = [{day:1,timeStart:1,timeEnd:2},{day:2,timeStart:1,timeEnd:2},{day:7,timeStart:1,timeEnd:2}];
//day：星期几，1 星期一，   timeStart:1 1点开始， timeEnd:几点结束
